    // This function creates a table row for a new Evidence Profile Stream within the Streams table in the form
    // This can be used to either create a new, empty stream, or add one that already exists within the Evidence Profile
    static createStreamRow(table_id, stream={}) {
        if (
            (typeof(table_id) == "string")
            && (table_id != "")
            && (typeof(stream) == "object")
            && (typeof(streamTypes) == "object")
            && (typeof(streamTypeValues) == "object")
            && (typeof(confidenceJudgements) == "object")
            && (typeof(confidenceJudgementValues) == "object")
        ) {
            // The incoming arguments match the desired types, the table_id is not empty and the global objects streamTypes and streamTypeValues
            // exist, continue

            // Get the tbody of the table indicated by the table_id argument
            var $tbody = $("#" + table_id + " tbody");
            if ($tbody.length > 0) {
                // The desired table body was found, add a cross-stream inferences row to it

                // Iterate through the existing rows in $tbody to come up with the index value to use when creating the next row
                var index = 0;
                var count = 0;
                $tbody.find("tr").each(
                    function(i, row) {
                        var $row = $(row);

                        var id = $row.attr("id");
                        if ((typeof(id) != "undefined") && (id != "") && (id.match(/^row_id_stream_\d+$/))) {
                            // The row object as an id attribute that matches the desired naming convention, retain the numeric
                            // portion if it is greater than the exiting index value
                            // Also, increment count by one to keep track of the number of valid rows found
                            index = Math.max(index, id.replace(/^row_id_stream_(\d+)$/, "$1"));
                            count++;
                        }
                    }
                );

                // Increment index and count by one to get the next values
                index++;
                count++;

                // Since this Boolean check will be needed multiple times, go ahead and do it once to use with all other times needed
                var hasConfidenceJudgement = ("confidence_judgement" in stream) && (typeof(stream.confidence_judgement) == "object");

                // Set the initial form field values based on the incoming stream object
                var stream_type = (("stream_type" in stream) && (typeof(stream.stream_type) == "number") && (streamTypeValues.includes(stream.stream_type))) ? stream.stream_type : -1;
                var stream_title = (("stream_title" in stream) && ((typeof(stream.stream_title) == "string") || (typeof(stream.stream_title) == "number"))) ? stream.stream_title : "";
                var confidence_judgement_title = ((hasConfidenceJudgement) && ("title" in stream.confidence_judgement) && ((typeof(stream.confidence_judgement.title) == "string") || (typeof(stream.confidence_judgement.title) == "number"))) ? stream.confidence_judgement.title : "";
                var confidence_judgement_score = ((hasConfidenceJudgement) && ("score" in stream.confidence_judgement) && (typeof(stream.confidence_judgement.score) == "number") && (confidenceJudgementValues.includes(stream.confidence_judgement.score))) ? stream.confidence_judgement.score : -1;
                var confidence_judgement_explanation = ((hasConfidenceJudgement) && ("explanation" in stream.confidence_judgement) && ((typeof(stream.confidence_judgement.explanation) == "string") || (typeof(stream.confidence_judgement.explanation) == "number"))) ? stream.confidence_judgement.explanation : "";

                var tr = '';
                tr = tr + '<tr id="row_id_stream_' + index + '">';
                tr = tr + ' <td style="border:0; vertical-align:top;">';
                tr = tr + '     <input type="hidden" id="stream_order_' + index + '" name="stream_order_' + index + '" class="orderingField" value="' + count + '" />';
                tr = tr + '     <select id="stream_type_' + index + '" name="stream_type_' + index + '">';
                tr = tr + '         <option ' + ((stream_type == -1) ? 'selected="selected" ' : '') + 'value="">--- Select Type ---</option>'

                for (var i=0; i<streamTypes.length; i++) {
                    tr = tr + '         <option ' + ((stream_type == streamTypes[i].value) ? 'selected="selected" ' : '') + 'value="' + streamTypes[i].value + '">' + streamTypes[i].name + '</option>';
                }

                tr = tr + '     </select>';
                tr = tr + ' </td>';
                tr = tr + ' <td style="border:0; vertical-align:top;"><input id="stream_title_' + index + '" class="span12 textinput textInput" type="text" maxlength="50" name="stream_title_' + index + '" required="required" value="' + stream_title + '" /></td>';

                tr = tr + ' <td style="border:0; vertical-align:top;"><input id="confidence_judgement_title_' + index + '" class="span12 textinput textInput" type="text" maxlength="50" name="confidence_judgement_title_' + index + '" required="required" value="' + confidence_judgement_title + '" /></td>';
                tr = tr + ' <td style="border:0; vertical-align:top;">';
                tr = tr + '     <select id="confidence_judgement_score_' + index + '" name="confidence_judgement_score_' + index + '">';
                tr = tr + '         <option ' + ((confidence_judgement_score == -1) ? 'selected="selected" ' : '') + 'value="">--- Select Score ---</option>';

                for (var i=0; i<confidenceJudgements.length; i++) {
                    tr = tr + '         <option ' + ((confidence_judgement_score == confidenceJudgements[i].fields.value) ? 'selected="selected" ' : '') + 'value="' + confidenceJudgements[i].fields.value + '">' + confidenceJudgements[i].fields.name + '</option>';
                }

                tr = tr + '     </select>';
                tr = tr + ' </td>';
                tr = tr + ' <td style="border:0; vertical-align:top;"><textarea id="confidence_judgement_explanation_' + index + '" class="span12" name="confidence_judgement_explanation_' + index + '" cols="40" rows="4" required="required">' + confidence_judgement_explanation + '</textarea></td>';
                tr = tr + ' <td id="outcomes_' + index + '" style="border:0; vertical-align:top;"></td>';
                tr = tr + ' <td style="border:0; vertical-align:top;">';
                tr = tr + '     <button id="btn_stream_moveup_' + index + '" class="btn btn-mini" title="move up" type="button"><i class="icon-arrow-up" /></button><br />';
                tr = tr + '     <button id="btn_stream_movedown_' + index + '" class="btn btn-mini" title="move down" type="button"><i class="icon-arrow-down" /></button><br />';
                tr = tr + '     <button id="btn_stream_remove_' + index + '" class="btn btn-mini" title="remove" type="button"><i class="icon-remove" /></button><br />';
                tr = tr + ' </td>';
                tr = tr + '</tr>';

                // Append the table row, set the click() functionality for the new buttons,  and then adjust the table's look and
                // variables' values as needed
                $tbody.append(tr);
                EvidenceProfile.buildOutcomeFormset(index);
                EvidenceProfile.setButtonSetClickActions("stream", index, EvidenceProfile.updateStreamsTable);
                EvidenceProfile.updateStreamsTable();
            }
        }
    }

    // This function sets the numbering and shading for the table holding the Evidence Profile Streams
    static updateStreamsTable() {
        // Iterate through each row within the table's body and set each stream's order
        $("#table_id_streams tbody").find("tr").each(
            function(i, row) {
                var $row = $(row);

                // Find the form field in the row that contains the stream's desired order (it has a class named "orderingField")
                $row.find(".orderingField").each(
                    function(j, field) {
                        $(field).val(i + 1);
                    }
                );

                // Set the row's color to either white (odd-numbered) or gray (even-numbered)
                $row.css("background-color", "#" + (((i % 2) == 1) ? "F1F1F1" : "FFFFFF"));
            }
        );
    }

    // This function sets the numbering and shading for the table holding the Cross-Stream Inferences
    static updateCrossStreamInferencesTable() {
        // Iterate through each row within the table's body and set the names for each form field contined within the row (there is
        // a portion of the name that indicates ordering, that is what gets changed)

        $("#table_id_inferences tbody").find("tr").each(
            function(i, row) {
                var $row = $(row);

                $row.find("input,textarea").each(
                    // Iterate over each form field found within the row and set the last portion of its name to the desired
                    // order value
                    function(j, field) {
                        $(field).attr("name", $(field).attr("name").replace(/\d+$/, (i + 1)));
                    }
                );

                // Set the row's color to either white (odd-numbered) or gray (even-numbered)
                $row.css("background-color", "#" + (((i % 2) == 1) ? "F1F1F1" : "FFFFFF"));
            }
        );
    }
